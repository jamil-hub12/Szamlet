# Comando para ejecutar los test en la terminal

pnpm vitest run test/[nombre del archivo]

# Ejemplo para ejecutar 1 por 1:

pnpm vitest run test/ficha01_validacionesCliente.test.ts

# Ejemplo de ejecución de todos los test:

pnpm vitest run

# En caso que no se tenga el pnpm instalado se puede utilizar el npm al inicio de cada comando
